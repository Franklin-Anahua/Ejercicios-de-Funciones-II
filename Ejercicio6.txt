6. Un banco ha decidido aumentar el límite de crédito de las tarjetas de crédito de sus clientes. Si el cliente tiene una tarjeta tipo 1, el aumento será del 25%; si es tipo 2, el aumento será del 35%; si es tipo 3, el aumento será del 40%; y para cualquier otro tipo, el aumento será del 50%. Genere un algoritmo donde se ingrese el tipo de tarjeta y el crédito actual, luego mostrar el nuevo crédito aumentado. Haga uso de una función que, dado el tipo de tarjeta y el monto, calcule el crédito aumentado.

#include <iostream>
using namespace std;

double f ( int tt, int ca ) {
    switch(tt) {
        case 1:
            return ca * 1.25;
            break;
        case 2:
            return ca * 1.35;
            break;
        case 3:
            return ca * 1.40;
            break;
        default:
            return ca * 1.50;
    }
}

int main() {
    int tt, ca;
    cout << "Ingrese el tipo de tarjeta: ";
    cin >> tt;
    cout << "Ingrese el crédito actual: ";
    cin >> ca;
    cout << "Tu nuevo credito aumentado es " << f(tt, ca) << endl;
    return 0;
}
