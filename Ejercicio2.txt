2. Si las vocales se representan con números del 1 al 5 (1 para 'a', 2 para 'e', 3 para 'i', 4 para 'o', 5 para 'u'), lea un número e indique qué vocal es. Haga uso de funciones para que, dado un número, retorne la vocal correspondiente.

#include <iostream>
using namespace std;

char encontrarVocal(int numero) {
    char vocal;
    switch(numero) {
        case 1:
            vocal = 'a';
            break;
        case 2:
            vocal = 'e';
            break;
        case 3:
            vocal = 'i';
            break;
        case 4:
            vocal = 'o';
            break;
        case 5:
            vocal = 'u';
            break;
        default:
            vocal = ' ';
    }
    return vocal;
}

int main() {
    int numero;
    cout << "Ingrese un número del 1 al 5: ";
    cin >> numero;

    char vocal = encontrarVocal(numero);

    if(vocal != ' ') {
        cout << "La vocal correspondiente al número " << numero << " es: " << vocal << endl;
    } else {
        cout << "Número inválido. Por favor ingrese un número del 1 al 5." << endl;
    }

    return 0;
}
