4. Cree una función que tome tres números como parámetros y devuelva el mayor de ellos.

#include <iostream>
using namespace std;

int mayo(int num1, int num2, int num3){
    if (num1 >= num2 && num1 >= num3){
        return num1;
    } else {
        if (num2 >= num1 && num2 >= num3){
            return num2;
        }
        return num3;
    }
}

int main(){
    int num1, num2, num3;
    cout << "Escrie tres numeros: " << endl;
    cin >> num1;
    cin >> num2;
    cin >> num3;
    int Mayor =  mayo(num1, num2, num3);
    cout << "El numero mayor es: " << Mayor << endl;
}
