1. Un alumno desea saber cuál será su calificación final en la primera unidad en el curso de fundamentos de programación. La calificación final se compone de los siguientes porcentajes: 40% de Evidencia de conocimiento, 35% de Evidencia de desempeño, y 25% de Evidencia de producto. Haga uso de funciones para el cálculo de la calificación final.

#include <iostream>
using namespace std;

double cal( int co, int des, int pro ){
    return co * 0.4 + des * 0.35 + pro * 0.25;
}
int main()
{
    int co, des, pro;
    cout << "Ingrese la calificación de la Evidencia de Conocimiento: ";
    cin >> co;
    cout << "Ingrese la calificación de la Evidencia de Desempeño: ";
    cin >> des;
    cout << "Ingrese la calificación de la Evidencia de Producto: ";
    cin >> pro;
    cout << "La calificación final es: " << cal( co, pro, des ) << endl;
    return 0;
}
